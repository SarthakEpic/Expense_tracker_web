from fastapi import FastAPI, Request, Form
from fastapi.templating import Jinja2Templates
from fastapi.responses import HTMLResponse, RedirectResponse
from fastapi.staticfiles import StaticFiles

import csv
import os
from datetime import datetime
from collections import defaultdict

import matplotlib.pyplot as plt

app = FastAPI()

# Static and template folders
app.mount("/static", StaticFiles(directory="static"), name="static")
templates = Jinja2Templates(directory="templates")

CSV_FILE = "expenses.csv"

# ✅ Get monthly totals from CSV
def get_monthly_totals():
    monthly_totals = defaultdict(float)
    try:
        with open(CSV_FILE, mode='r') as file:
            reader = csv.reader(file)
            for row in reader:
                amount, category, description, date_str = row
                date_obj = datetime.strptime(date_str, "%Y-%m-%d %H:%M:%S")
                month = date_obj.strftime("%Y-%m")
                monthly_totals[month] += float(amount)
    except FileNotFoundError:
        pass

    return sorted(monthly_totals.items())

# ✅ Generate bar chart
def generate_monthly_chart(monthly_totals):
    months = [month for month, _ in monthly_totals]
    totals = [total for _, total in monthly_totals]

    plt.figure(figsize=(8, 4))
    plt.bar(months, totals, color="#2e86de")
    plt.xlabel("Month")
    plt.ylabel("Total Spending (₹)")
    plt.title("Monthly Spending")
    plt.xticks(rotation=45)
    plt.tight_layout()

    chart_path = os.path.join("static", "monthly_chart.png")
    plt.savefig(chart_path)
    plt.close()

    return "/static/monthly_chart.png"

# ✅ Home route — show form + expenses + chart
@app.get("/", response_class=HTMLResponse)
async def home(request: Request, message: str = ""):
    expenses = []
    try:
        with open(CSV_FILE, mode='r') as file:
            reader = csv.reader(file)
            expenses = list(reader)
    except FileNotFoundError:
        pass

    monthly_totals = get_monthly_totals()
    chart_url = generate_monthly_chart(monthly_totals)

    return templates.TemplateResponse("index.html", {
        "request": request,
        "message": message,
        "expenses": expenses,
        "monthly_totals": monthly_totals,
        "chart_url": chart_url
    })

# ✅ Add new expense
@app.post("/add")
async def add_expense(
    request: Request,
    amount: float = Form(...),
    category: str = Form(...),
    description: str = Form("")
):
    date = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    with open(CSV_FILE, mode='a', newline='') as file:
        writer = csv.writer(file)
        writer.writerow([amount, category, description, date])

    message = "✅ Expense added successfully!"

    expenses = []
    with open(CSV_FILE, mode='r') as file:
        reader = csv.reader(file)
        expenses = list(reader)

    monthly_totals = get_monthly_totals()
    chart_url = generate_monthly_chart(monthly_totals)

    return templates.TemplateResponse("index.html", {
        "request": request,
        "message": message,
        "expenses": expenses,
        "monthly_totals": monthly_totals,
        "chart_url": chart_url
    })

# ✅ Delete expense by index
@app.get("/delete/{index}")
async def delete_expense(index: int):
    try:
        with open(CSV_FILE, mode='r') as file:
            expenses = list(csv.reader(file))

        if 0 <= index < len(expenses):
            expenses.pop(index)

            with open(CSV_FILE, mode='w', newline='') as file:
                writer = csv.writer(file)
                writer.writerows(expenses)
    except Exception as e:
        print("Error deleting expense:", e)

    return RedirectResponse(url="/", status_code=302)

# Edit expense
@app.get("/edit/{index}", response_class=HTMLResponse)
async def edit_expense_form(request: Request, index: int):
    try:
        with open(CSV_FILE, mode='r') as file:
            expenses = list(csv.reader(file))

        if 0 <= index < len(expenses):
            expense = expenses[index]
            return templates.TemplateResponse("edit.html", {
                "request": request,
                "index": index,
                "amount": expense[0],
                "category": expense[1],
                "description": expense[2]
            })
    except Exception as e:
        print("Error loading edit form:", e)

    return RedirectResponse(url="/", status_code=302)

# Save edit expense
@app.post("/edit/{index}")
async def save_edited_expense(
    request: Request,
    index: int,
    amount: float = Form(...),
    category: str = Form(...),
    description: str = Form("")
):
    try:
        with open(CSV_FILE, mode='r') as file:
            expenses = list(csv.reader(file))

        if 0 <= index < len(expenses):
            # Keep the original date unchanged
            old_date = expenses[index][3]
            expenses[index] = [amount, category, description, old_date]

            with open(CSV_FILE, mode='w', newline='') as file:
                writer = csv.writer(file)
                writer.writerows(expenses)

    except Exception as e:
        print("Error saving edited expense:", e)

    return RedirectResponse(url="/", status_code=302)
